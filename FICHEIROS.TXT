struct configuration read_conf(){
 	FILE *f = fopen(CONF_FILE,"r");
 	

 	struct configuration conf_struct;
 	if(f)
 		printf("Path is valid\n");
 	else{
 		printf("\nPath is not valid\n");
 		system("pause");
 		exit(0);
 	}

 	if(f!=NULL){
 		int i=1;
 		char line[SIZE_BUF];
 		printf("Reading configurations.txt file\n\n");
 		
 		while(fgets(line,sizeof(line),f)!=NULL){
 			char *temp;
 			temp = strstr((char *)line,DELIMITATOR);
 			temp = temp + strlen(DELIMITATOR);
 			
 			switch(i){
 				case 1:
 				memcpy(conf_struct.port,temp,strlen(temp));
 				break;
 				case 2:
 				memcpy(conf_struct.n,temp,strlen(temp));
 				break;
 				case 3:
 				memcpy(conf_struct.policy,temp,strlen(temp));
 				break;
 				case 4:
 				memcpy(conf_struct.script,temp,strlen(temp));
 				break;

 			}
 			i++;
 		}
 	}
 	else{
 		printf("No configurations file found\n");
 		f=fopen(CONF_FILE,"w");
 	}
 	fclose(f);
 	return conf_struct;
 }
